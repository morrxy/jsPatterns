function Waffle() {
	if (!(this instanceof Waffle)) {
		return new Waffle();
	}

	this.tastes = "yummy";
}
Waffle.prototype.wantAnother = true;

// testing invocations
var first = new Waffle(),
	second = Waffle();

console.log(first.tastes);
console.log(second.tastes);

console.log(first.wantAnother);
console.log(second.wantAnother);
