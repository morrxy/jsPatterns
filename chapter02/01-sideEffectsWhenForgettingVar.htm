side effects when forgetting var

from book JavaScript Patterns pdf version, page 30

globals created with var cannot be deleteed.

implied globals created without var can be deleted.

define three globals
var global_var = 1;
global_novar = 2; // antipattern
(function () {
	global_fromfunc = 3; // antipattern
})();

// attemp to delete
console.log(delete global_var);
console.log(delete global_novar);
console.log(delete global_fromfunc);

// test the deletion
console.log(typeof global_var);
console.log(typeof global_novar);
console.log(typeof global_fromfunc);
